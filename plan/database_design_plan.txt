Below is a production-grade relational database design (PostgreSQL) tailored for your textile e-commerce system. This is not a toy schema—this is structured for scalability, analytics, and future AI integration.

🧠 1. DATABASE DESIGN PRINCIPLES
Normalized (3NF) → avoids redundancy
Extensible → supports future features (AI, analytics, multi-vendor)
Audit-ready → timestamps + tracking
Secure → no sensitive data exposure
🧩 2. CORE ENTITY OVERVIEW
4
🗄️ 3. TABLES (DETAILED DESIGN)
👤 3.1 USERS TABLE
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email VARCHAR(255) UNIQUE NOT NULL,
    phone VARCHAR(20) UNIQUE,
    password_hash TEXT NOT NULL,
    is_verified BOOLEAN DEFAULT FALSE,
    status VARCHAR(20) DEFAULT 'active',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
🧾 3.2 USER PROFILE
CREATE TABLE user_profiles (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    first_name VARCHAR(100),
    last_name VARCHAR(100),
    gender VARCHAR(10),
    date_of_birth DATE,
    profile_image TEXT,
    preferred_language VARCHAR(10),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
🔐 3.3 LOGIN SESSIONS / AUTH TRACKING
CREATE TABLE login_sessions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    token TEXT NOT NULL,
    device_info TEXT,
    ip_address VARCHAR(50),
    expires_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
🌍 3.4 LOCALE / REGION TABLE
CREATE TABLE locales (
    id SERIAL PRIMARY KEY,
    country VARCHAR(100),
    state VARCHAR(100),
    city VARCHAR(100),
    currency VARCHAR(10),
    language VARCHAR(10)
);
📍 3.5 USER ADDRESSES
CREATE TABLE user_addresses (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    address_line1 TEXT,
    address_line2 TEXT,
    city VARCHAR(100),
    state VARCHAR(100),
    country VARCHAR(100),
    postal_code VARCHAR(20),
    is_default BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
🛍️ PRODUCT SYSTEM
🧵 3.6 PRODUCTS TABLE
CREATE TABLE products (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(255) NOT NULL,
    description TEXT,
    category_id UUID,
    base_price NUMERIC(10,2),
    brand VARCHAR(100),
    fabric_type VARCHAR(100),
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
🧾 3.7 PRODUCT VARIANTS (SIZE / COLOR)
CREATE TABLE product_variants (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    product_id UUID REFERENCES products(id) ON DELETE CASCADE,
    size VARCHAR(10),
    color VARCHAR(50),
    sku VARCHAR(100) UNIQUE,
    price NUMERIC(10,2),
    stock INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
🖼️ 3.8 PRODUCT IMAGES
CREATE TABLE product_images (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    product_id UUID REFERENCES products(id) ON DELETE CASCADE,
    image_url TEXT,
    is_primary BOOLEAN DEFAULT FALSE
);
🏷️ 3.9 CATEGORIES
CREATE TABLE categories (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(100),
    parent_id UUID REFERENCES categories(id)
);
🛒 ORDER SYSTEM
📦 3.10 ORDERS
CREATE TABLE orders (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id),
    total_amount NUMERIC(10,2),
    status VARCHAR(50),
    payment_status VARCHAR(50),
    shipping_address_id UUID,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
📄 3.11 ORDER ITEMS
CREATE TABLE order_items (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    order_id UUID REFERENCES orders(id) ON DELETE CASCADE,
    product_variant_id UUID REFERENCES product_variants(id),
    quantity INT,
    price NUMERIC(10,2)
);
💳 3.12 PAYMENTS
CREATE TABLE payments (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    order_id UUID REFERENCES orders(id),
    payment_gateway VARCHAR(50),
    transaction_id VARCHAR(255),
    amount NUMERIC(10,2),
    status VARCHAR(50),
    paid_at TIMESTAMP
);
🧠 OPTIONAL (BUT IMPORTANT)
❤️ 3.13 WISHLIST
CREATE TABLE wishlists (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id),
    product_id UUID REFERENCES products(id)
);
🛒 3.14 CART (SERVER SIDE)
CREATE TABLE carts (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id)
);

CREATE TABLE cart_items (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    cart_id UUID REFERENCES carts(id),
    product_variant_id UUID REFERENCES product_variants(id),
    quantity INT
);
⭐ 3.15 REVIEWS
CREATE TABLE reviews (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id),
    product_id UUID REFERENCES products(id),
    rating INT CHECK (rating BETWEEN 1 AND 5),
    comment TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
🔗 4. RELATIONSHIP SUMMARY
User → Profile (1:1)
User → Orders (1:N)
Order → Items (1:N)
Product → Variants (1:N)
Product → Images (1:N)
Variant → Order Items (1:N)
⚡ 5. INDEXING (CRITICAL FOR PERFORMANCE)
CREATE INDEX idx_products_name ON products(name);
CREATE INDEX idx_orders_user_id ON orders(user_id);
CREATE INDEX idx_variants_product_id ON product_variants(product_id);
🔐 6. SECURITY BEST PRACTICES
Store only hashed passwords (bcrypt)
Never expose tokens directly
Use row-level security (future)
